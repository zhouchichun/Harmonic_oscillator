\[
L=\operatorname{Im}\left[  \psi^{\dagger}\left(  x,t\right)  \dot{\psi}\left(
x,t\right)  \right]  -\frac{1}{2}\nabla\psi^{\dagger}\left(  x,t\right)
\nabla\psi\left(  x,t\right)  +V\left(  x\right)  \psi^{\dagger}\left(
x,t\right)  \psi\left(  x,t\right)
\]
令\[
\psi\left(  x,t\right)  =u\left(  x,t\right)  +iv\left(  x,t\right)
\]
第一部分\begin{align*}
\psi^{\dagger}\left(  x,t\right)  \dot{\psi}\left(  x,t\right)   &  =\left[
u\left(  x,t\right)  -iv\left(  x,t\right)  \right]  \left[  \dot{u}\left(
x,t\right)  +i\dot{v}\left(  x,t\right)  \right] \\
&  =u\left(  x,t\right)  \dot{u}\left(  x,t\right)  +iu\left(  x,t\right)
\dot{v}\left(  x,t\right)  -iv\left(  x,t\right)  \dot{u}\left(  x,t\right)
+v\left(  x,t\right)  \dot{v}\left(  x,t\right) \\
&  =u\left(  x,t\right)  \dot{u}\left(  x,t\right)  +v\left(  x,t\right)
\dot{v}\left(  x,t\right)  +i\left[  u\left(  x,t\right)  \dot{v}\left(
x,t\right)  -v\left(  x,t\right)  \dot{u}\left(  x,t\right)  \right]
\end{align*}
第二部分\begin{align*}
\nabla\psi^{\dagger}\left(  x,t\right)  \nabla\psi\left(  x,t\right)   &
=\left[  u^{\prime}\left(  x,t\right)  -iv^{\prime}\left(  x,t\right)
\right]  \left[  u^{\prime}\left(  x,t\right)  +iv^{\prime}\left(  x,t\right)
\right] \\
&  =u^{\prime2}\left(  x,t\right)  +v^{\prime2}\left(  x,t\right)
\end{align*}
第三部分\[
\psi^{\dagger}\left(  x,t\right)  \psi\left(  x,t\right)  =u^{2}\left(
x,t\right)  +v^{2}\left(  x,t\right)
\]
最后写为$u$和$v$的形式
\begin{align*}
L  &  =u\left(  x,t\right)  \dot{v}\left(  x,t\right)  -v\left(  x,t\right)
\dot{u}\left(  x,t\right) \\
&  -\frac{1}{2}u^{\prime2}\left(  x,t\right)  -\frac{1}{2}v^{\prime2}\left(
x,t\right) \\
&  +V\left(  x\right)  u^{2}\left(  x,t\right)  +V\left(  x\right)
v^{2}\left(  x,t\right)
\end{align*}


作用量为\begin{align*}
S  &  =\int dxdtL\\
&  =\int dxdtu\left(  x,t\right)  \dot{u}\left(  x,t\right)  +\int
dxdtv\left(  x,t\right)  \dot{v}\left(  x,t\right) \\
&  -\frac{1}{2}\int dxdtu^{\prime2}\left(  x,t\right)  -\frac{1}{2}\int
dxdtv^{\prime2}\left(  x,t\right) \\
&  +\int dxdtV\left(  x\right)  u^{2}\left(  x,t\right)  +\int dxdtV\left(
x\right)  v^{2}\left(  x,t\right)
\end{align*}
